/* 11. :focus pseudo-class (use list selector in this exact order) */
input:focus,
textarea:focus {
  border-color: #007bff; /* Blue border on focus */
  outline: none;
}

/* 12. :invalid pseudo-class (exact order: input, select, textarea) */
input:invalid,
select:invalid,
textarea:invalid {
  border-color: red;
}

/* 13. :valid pseudo-class (exact order: input, select, textarea) */
input:valid,
select:valid,
textarea:valid {
  border-color: green;
}

/* 14. :hover pseudo-class for button */
button:hover {
  background-color: #0056b3; /* Changes color on hover */
  cursor: pointer;
}

/* 15â€“17. :checked pseudo-class for .radio-group radio buttons */
.radio-group input[type="radio"]:checked {
  border-color: #007bff;      /* Adds border color */
  background-color: #cce5ff;  /* Adds background color */
  box-shadow: 0 0 5px #007bff; /* Adds box shadow */
}

/* 18. :checked + label changes label color when selected */
.radio-group input[type="radio"]:checked + label {
  color: #007bff;
}

/* 19. :first-of-type pseudo-class on first input */
input:first-of-type {
  border-radius: 10px; /* Example of styling the first input differently */
  border: 2px solid orange;
}
